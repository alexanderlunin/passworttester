        <button onclick="checkPassword()">Passwort testen</button>

        <p id="result"></p>

    <script>

        function checkPassword() {

            const password = document.getElementById('passwordInput').value;

            const result = document.getElementById('result');

            let score = 0;


            // Bewertungskriterien

            if (password.length >= 8) score++; // Mindestlänge

            if (/[A-Z]/.test(password)) score++; // Großbuchstaben

            if (/[a-z]/.test(password)) score++; // Kleinbuchstaben

            if (/[0-9]/.test(password)) score++; // Zahlen

            if (/[^A-Za-z0-9]/.test(password)) score++; // Sonderzeichen


            // Bewertung anzeigen

            if (password.length === 0) {

                result.textContent = "Bitte geben Sie ein Passwort ein.";

                result.style.color = "black";

            } else if (score <= 2) {

                result.textContent = "Unsicheres Passwort.";

                result.style.color = "red";

            } else if (score === 3 || score === 4) {

                result.textContent = "Mittleres Passwort.";

                result.style.color = "orange";

            } else {

                result.textContent = "Sicheres Passwort!";

                result.style.color = "green";

            }

        }

    </script>
